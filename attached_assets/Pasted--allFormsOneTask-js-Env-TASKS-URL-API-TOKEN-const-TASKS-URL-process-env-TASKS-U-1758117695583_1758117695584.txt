// Env: TASKS_URL, API_TOKEN

const TASKS_URL = process.env.TASKS_URL;
const API_TOKEN  = process.env.API_TOKEN;

// Allowed step types (lowercase only)
const TYPES = {
  SYS: 'system_action',
  COMM: 'communication',
  DOCS: 'documentation',
  VERIFY: 'verification',
  INSP: 'inspection',
  APPROVAL: 'approval',
  SECURITY: 'security',
  RECEIVED: 'received'
};

// ---- MASTER: one-and-only-one assignment creator (upsert) ----
async function createOneAssignment({ externalId, title, labels = [], checklist }) {
  const payload = {
    external_id: externalId,           // server can use this to upsert
    title,
    priority: 'high',
    labels,
    source: 'form',                    // <- tells other listeners to SKIP
    task_mode: 'embedded',             // <- extra guard
    checklist
  };

  // Prefer upsert endpoint if you have it; otherwise use /assignments with Idempotency-Key
  const url = `${TASKS_URL}/assignments.upsert`;
  const res = await fetch(url, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${API_TOKEN}`,
      'Idempotency-Key': externalId     // belt + suspenders
    },
    body: JSON.stringify(payload)
  });

  if (!res.ok) {
    const txt = await res.text().catch(() => '');
    throw new Error(`Assignment upsert failed (${res.status}): ${txt}`);
  }
  return res.json().catch(() => ({}));
}

// ---- ROUTER: handle ALL form types with one call ----
// Expected form fields:
// - form.type: 'onboarding_fleet_day0' | 'onboarding_assets_day0' | 'onboarding_assets_day1_5' | 'onboarding_inventory' | 'offboarding_day0' | 'byov_onboarding_fleet' | 'assign_vehicle' | 'create_vehicle'
// - plus IDs like employee_id, vehicle_id, dates used to build keys.
async function handleFormSubmit(form) {
  const type = String(form.type || '').toLowerCase();

  // Build a unique key per submission (prevents duplicates across ALL forms)
  const keyBits = [
    type,
    form.employee_id || '',
    form.vehicle_id || form.license_plate || form.vin || '',
    form.start_date || form.last_day || form.date || ''
  ].filter(Boolean);

  const externalId = keyBits.join(':');

  // Route to the right checklist
  switch (type) {
    case 'byov_onboarding_fleet': {
      const title = `BYOV Onboarding — Fleet (${form.employee_name || form.submitter || 'Technician'})`;
      return createOneAssignment({
        externalId,
        title,
        labels: ['byov','onboarding','fleet'],
        checklist: checklist_byov_fleet()
      });
    }

    case 'onboarding_fleet_day0': {
      const title = `Onboarding Day 0 — Fleet (${form.employee_name || 'Technician'})`;
      return createOneAssignment({
        externalId,
        title,
        labels: ['onboarding','fleet','day0'],
        checklist: checklist_onboarding_day0_fleet()
      });
    }

    case 'onboarding_assets_day0': {
      const title = `Onboarding Day 0 — Assets (${form.employee_name || 'Technician'})`;
      return createOneAssignment({
        externalId,
        title,
        labels: ['onboarding','assets','day0'],
        checklist: checklist_onboarding_day0_assets()
      });
    }

    case 'onboarding_assets_day1_5': {
      const title = `Onboarding Day 1–5 — Assets (${form.employee_name || 'Technician'})`;
      return createOneAssignment({
        externalId,
        title,
        labels: ['onboarding','assets','day1-5'],
        checklist: checklist_onboarding_day1_5_assets()
      });
    }

    case 'onboarding_inventory': {
      const title = `Onboarding — Inventory (${form.employee_name || 'Technician'})`;
      return createOneAssignment({
        externalId,
        title,
        labels: ['onboarding','inventory'],
        checklist: checklist_onboarding_inventory()
      });
    }

    // Example offboarding — if you already have a Day 0 template, plug it here
    case 'offboarding_day0': {
      const title = `Offboarding Day 0 — ${form.employee_name || 'Technician'}`;
      // TODO: swap the checklist builder below to your offboarding Day 0 steps
      return createOneAssignment({
        externalId,
        title,
        labels: ['offboarding','day0'],
        checklist: [
          { id:'ntao_hold', title:'Place shipping hold', type: TYPES.SYS },
          { id:'ntao_cancel', title:'Cancel pending & backorders', type: TYPES.SYS },
          { id:'inv_tpms', title:'Remove truck from TPMS', type: TYPES.SYS },
          { id:'fleet_contact', title:'Contact tech to schedule retrieval', type: TYPES.COMM },
          { id:'equip_reconcile', title:'Reconcile tools & iPhone', type: TYPES.VERIFY }
        ]
      });
    }

    // Optional: create/assign vehicle forms (kept simple)
    case 'assign_vehicle': {
      const title = `Assign/Update Vehicle — ${form.employee_name || ''}`.trim();
      return createOneAssignment({
        externalId,
        title,
        labels: ['vehicle','assign'],
        checklist: [
          { id:'dash_update', title:'Update fleet dashboard assignment', type: TYPES.SYS },
          { id:'notify', title:'Notify technician of assignment', type: TYPES.COMM },
          { id:'truce_update', title:'Update TRUCE', type: TYPES.SYS }
        ]
      });
    }

    case 'create_vehicle': {
      const title = `Create Vehicle — ${form.make || ''} ${form.model || ''} ${form.license_plate || form.vin || ''}`.trim();
      return createOneAssignment({
        externalId,
        title,
        labels: ['vehicle','create'],
        checklist: [
          { id:'ams_add', title:'Add vehicle to AMS/Holman', type: TYPES.DOCS },
          { id:'fleet_tag', title:'Assign fleet tag & dashboard status', type: TYPES.SYS },
          { id:'photos', title:'Capture initial photos', type: TYPES.DOCS }
        ]
      });
    }

    default:
      throw new Error(`Unknown form.type: "${form.type}". Please pass one of the supported types.`);
  }
}

// ---- LISTENER GUARD (put this once where your generic rules run) ----
function shouldSkip(evt) {
  return evt?.source === 'form' || evt?.task_mode === 'embedded';
}

// Example: generic onboarding/offboarding listeners

on('offboarding_event', async (evt) => { if (shouldSkip(evt)) return; /* generic logic for non-form sources */ });

module.exports = { handleFormSubmit };
