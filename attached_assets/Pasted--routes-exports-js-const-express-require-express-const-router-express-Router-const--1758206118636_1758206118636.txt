// routes/exports.js
const express = require('express');
const router = express.Router();
const { Pool } = require('pg');
const { stringify } = require('csv-stringify/sync');
const ExcelJS = require('exceljs');

const pool = new Pool({ connectionString: process.env.DATABASE_URL });

function toArray(v){ if(!v) return null; return Array.isArray(v)?v:[v]; }

function buildWhere(q) {
  const clauses = ['created_at >= $1', 'created_at < $2'];
  const binds = [q.from_ts, q.to_ts];
  let idx = 3;

  if (q.departments) { clauses.push(`department = ANY($${idx})`); binds.push(q.departments); idx++; }
  if (q.statuses)    { clauses.push(`status = ANY($${idx})`);     binds.push(q.statuses);    idx++; }
  if (q.assignees)   { clauses.push(`assignee = ANY($${idx})`);   binds.push(q.assignees);   idx++; }

  return { where: 'WHERE ' + clauses.join(' AND '), binds };
}

const BASE_SELECT = `
  SELECT id, department, team, type, status, assignee,
         created_at, started_at, first_response_at, completed_at, updated_at
  FROM requests
`;

router.get('/exports/requests.csv', async (req, res, next) => {
  try {
    const q = {
      from_ts: req.query.from_ts, to_ts: req.query.to_ts,
      departments: toArray(req.query.departments),
      statuses: toArray(req.query.statuses),
      assignees: toArray(req.query.assignees),
    };
    const { where, binds } = buildWhere(q);
    const { rows } = await pool.query(`${BASE_SELECT} ${where} ORDER BY created_at DESC`, binds);
    const csv = rows.length ? stringify(rows, { header: true }) : '';
    res.setHeader('Content-Type', 'text/csv');
    res.setHeader('Content-Disposition', 'attachment; filename=requests.csv');
    res.send(csv);
  } catch (e) { next(e); }
});

router.get('/exports/requests.xlsx', async (req, res, next) => {
  try {
    const q = {
      from_ts: req.query.from_ts, to_ts: req.query.to_ts,
      departments: toArray(req.query.departments),
      statuses: toArray(req.query.statuses),
      assignees: toArray(req.query.assignees),
    };
    const { where, binds } = buildWhere(q);
    const { rows } = await pool.query(`${BASE_SELECT} ${where} ORDER BY created_at DESC`, binds);

    const wb = new ExcelJS.Workbook();
    const ws = wb.addWorksheet('Requests');
    if (rows.length) {
      ws.columns = Object.keys(rows[0]).map(k => ({ header: k, key: k }));
      rows.forEach(r => ws.addRow(r));
    }
    res.setHeader('Content-Type', 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet');
    res.setHeader('Content-Disposition', 'attachment; filename=requests.xlsx');
    await wb.xlsx.write(res);
    res.end();
  } catch (e) { next(e); }
});

module.exports = router;
