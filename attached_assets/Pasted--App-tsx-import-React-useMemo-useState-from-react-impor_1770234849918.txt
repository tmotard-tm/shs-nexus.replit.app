```App.tsx
import React, { useMemo, useState } from 'react'
import { mockRecoveryCases } from './utils/mockData'
import { ToolsRecoveryTable } from './components/ToolsRecoveryTable'
import { FilterBar } from './components/FilterBar'
import { Badge } from './components/ui/Badge'
import {
  LayoutDashboard,
  AlertTriangle,
  Clock,
  CheckCircle,
} from 'lucide-react'
export function App() {
  const [searchQuery, setSearchQuery] = useState('')
  const [filters, setFilters] = useState<{
    status: string[]
    vehicleType: string[]
    district: string[]
    incompleteOnly: boolean
  }>({
    status: [],
    vehicleType: [],
    district: [],
    incompleteOnly: false,
  })
  // Filter Logic
  const filteredData = useMemo(() => {
    return mockRecoveryCases.filter((item) => {
      const searchLower = searchQuery.toLowerCase()
      const matchesSearch =
        item.techName.toLowerCase().includes(searchLower) ||
        item.truckNumber.toLowerCase().includes(searchLower) ||
        item.routing.toLowerCase().includes(searchLower) ||
        item.enterpriseId.toLowerCase().includes(searchLower)
      const matchesStatus =
        filters.status.length === 0 || filters.status.includes(item.status)
      const matchesVehicle =
        filters.vehicleType.length === 0 ||
        filters.vehicleType.includes(item.vehicleType)
      const matchesDistrict =
        filters.district.length === 0 ||
        filters.district.includes(item.district)
      const matchesIncomplete =
        !filters.incompleteOnly ||
        Object.values(item.tasks).some((completed) => !completed)
      return (
        matchesSearch &&
        matchesStatus &&
        matchesVehicle &&
        matchesDistrict &&
        matchesIncomplete
      )
    })
  }, [searchQuery, filters])
  // Stats Calculation
  const stats = useMemo(() => {
    return {
      total: mockRecoveryCases.length,
      urgent: mockRecoveryCases.filter(
        (i) => i.vehicleType === 'BYOV' || i.vehicleType === 'Rental',
      ).length,
      inProgress: mockRecoveryCases.filter((i) => i.status === 'In Progress')
        .length,
      completed: mockRecoveryCases.filter((i) => i.status === 'Completed')
        .length,
    }
  }, [])
  // Handlers
  const handleFilterChange = (
    type: 'status' | 'vehicleType' | 'district' | 'incompleteOnly',
    value: any,
  ) => {
    if (type === 'incompleteOnly') {
      setFilters((prev) => ({
        ...prev,
        incompleteOnly: value,
      }))
    } else {
      setFilters((prev) => ({
        ...prev,
        [type]: value ? [value] : [],
      }))
    }
  }
  const handleClearFilters = () => {
    setSearchQuery('')
    setFilters({
      status: [],
      vehicleType: [],
      district: [],
      incompleteOnly: false,
    })
  }
  return (
    <div className="min-h-screen bg-slate-100 font-sans text-slate-900">
      {/* Header */}
      <header className="bg-white border-b border-slate-200 sticky top-0 z-20">
        <div className="max-w-7xl mx-auto px-6 h-14 flex items-center justify-between">
          <div className="flex items-center gap-3">
            <div className="bg-[#1A4B8C] p-1.5 rounded">
              <LayoutDashboard className="h-4 w-4 text-white" />
            </div>
            <h1 className="text-lg font-semibold text-[#1A4B8C]">
              Tools Recovery Queue
            </h1>
          </div>

          <div className="flex items-center gap-3">
            <span className="text-sm text-slate-500">{stats.total} cases</span>
            <div className="h-4 w-px bg-slate-200" />
            <Badge variant="destructive" className="h-6 px-2 gap-1 text-xs">
              <AlertTriangle className="h-3 w-3" />
              {stats.urgent} Urgent
            </Badge>
            <Badge variant="navy" className="h-6 px-2 gap-1 text-xs">
              <Clock className="h-3 w-3" />
              {stats.inProgress} Active
            </Badge>
            <Badge variant="teal" className="h-6 px-2 gap-1 text-xs">
              <CheckCircle className="h-3 w-3" />
              {stats.completed} Done
            </Badge>
          </div>
        </div>
      </header>

      {/* Main Content */}
      <main className="max-w-7xl mx-auto px-6 py-6">
        <FilterBar
          searchQuery={searchQuery}
          onSearchChange={setSearchQuery}
          activeFilters={filters}
          onFilterChange={handleFilterChange}
          onClearFilters={handleClearFilters}
        />

        <ToolsRecoveryTable data={filteredData} />
      </main>
    </div>
  )
}
```

```components/FilterBar.tsx
import React from 'react'
import { Search, X, ListTodo } from 'lucide-react'
interface FilterBarProps {
  searchQuery: string
  onSearchChange: (query: string) => void
  activeFilters: {
    status: string[]
    vehicleType: string[]
    district: string[]
    incompleteOnly: boolean
  }
  onFilterChange: (
    type: 'status' | 'vehicleType' | 'district' | 'incompleteOnly',
    value: any,
  ) => void
  onClearFilters: () => void
}
export function FilterBar({
  searchQuery,
  onSearchChange,
  activeFilters,
  onFilterChange,
  onClearFilters,
}: FilterBarProps) {
  const hasActiveFilters =
    activeFilters.status.length > 0 ||
    activeFilters.vehicleType.length > 0 ||
    activeFilters.district.length > 0 ||
    activeFilters.incompleteOnly ||
    searchQuery
  return (
    <div className="flex flex-wrap items-center gap-3 mb-4">
      {/* Search */}
      <div className="relative flex-1 min-w-[200px] max-w-sm">
        <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-slate-400" />
        <input
          type="text"
          placeholder="Search by name or ID..."
          value={searchQuery}
          onChange={(e) => onSearchChange(e.target.value)}
          className="w-full pl-9 pr-3 py-2 text-sm bg-white border border-slate-200 rounded-lg focus:outline-none focus:ring-2 focus:ring-[#1A4B8C]/20 focus:border-[#1A4B8C]"
        />

      {/* Filters */}
      <select
        className="h-9 text-sm bg-white border border-slate-200 rounded-lg px-3 focus:outline-none focus:ring-2 focus:ring-[#1A4B8C]/20"

        <option value="">All Vehicles</option>
        <option value="Company">Company</option>
        <option value="BYOV">BYOV</option>
        <option value="Rental">Rental</option>
      </select>

      <select
        className="h-9 text-sm bg-white border border-slate-200 rounded-lg px-3 focus:outline-none focus:ring-2 focus:ring-[#1A4B8C]/20"
        value={activeFilters.district[0] || ''}

        <ListTodo className="h-4 w-4" />
```

```components/ToolsRecoveryTable.tsx
import React, { useMemo, useState, Fragment } from 'react'
import { RecoveryCase } from '../types/recovery'
import { ExpandedRowDetails } from './ExpandedRowDetails'
import { Badge } from './ui/Badge'
import { AlertTriangle, ChevronDown, ChevronRight, Check } from 'lucide-react'
interface ToolsRecoveryTableProps {
  data: RecoveryCase[]
}
type SortConfig = {
  key: keyof RecoveryCase
  direction: 'asc' | 'desc'
} | null
export function ToolsRecoveryTable({ data }: ToolsRecoveryTableProps) {
  const [expandedRowId, setExpandedRowId] = useState<string | null>(null)
  const [sortConfig, setSortConfig] = useState<SortConfig>(null)
  const [currentPage, setCurrentPage] = useState(1)
  const itemsPerPage = 8
  const handleSort = (key: keyof RecoveryCase) => {
    let direction: 'asc' | 'desc' = 'asc'
    if (
      sortConfig &&
      sortConfig.key === key &&
      sortConfig.direction === 'asc'
    ) {
      direction = 'desc'
    }
    setSortConfig({
      key,
      direction,
    })
  }
  const sortedData = useMemo(() => {
    if (!sortConfig) return data
    return [...data].sort((a, b) => {
      if (a[sortConfig.key] < b[sortConfig.key]) {
        return sortConfig.direction === 'asc' ? -1 : 1
      }
      if (a[sortConfig.key] > b[sortConfig.key]) {
        return sortConfig.direction === 'asc' ? 1 : -1
      }
      return 0
    })
  }, [data, sortConfig])
  // Pagination
  const totalPages = Math.ceil(sortedData.length / itemsPerPage)
  const paginatedData = sortedData.slice(
    (currentPage - 1) * itemsPerPage,
    currentPage * itemsPerPage,
  )
  const toggleRowExpansion = (id: string) => {
    setExpandedRowId(expandedRowId === id ? null : id)
  }
  const getStatusBadgeVariant = (status: string) => {
    switch (status) {
      case 'Completed':
        return 'success'
      case 'Escalated':
        return 'destructive'
      case 'In Progress':
        return 'navy'
      case 'Contacted':
        return 'teal'
      default:
        return 'secondary'
    }
  }
  const isUrgentRow = (row: RecoveryCase) => {
    return row.vehicleType === 'BYOV' || row.vehicleType === 'Rental'
  }
  const getTaskProgress = (tasks: RecoveryCase['tasks']) => {
    const total = 6
    const completed = Object.values(tasks).filter(Boolean).length
    return {
      completed,
      total,
      percentage: (completed / total) * 100,
    }
  }
  const columns = [
    {
      key: 'techName',
      label: 'Technician',
      width: 'w-48',
    },
    {
      key: 'district',
      label: 'District',
      width: 'w-20',
    },
    {
      key: 'separationDate',
      label: 'Sep Date',
      width: 'w-28',
    },
    {
      key: 'vehicleType',
      label: 'Vehicle',
      width: 'w-24',
    },
    {
      key: 'routing',
      label: 'Routing',
      width: 'w-24',
    },
    {
      key: 'status',
      label: 'Status',
      width: 'w-28',
    },
    {
      key: 'tasks',
      label: 'Tasks',
      width: 'w-32',
    },
  ]
  return (
    <div className="w-full bg-white border border-slate-200 rounded-lg shadow-sm overflow-hidden">
      <div className="overflow-x-auto">
        <table className="w-full text-sm text-left">
          <thead className="bg-[#1A4B8C] text-white">
            <tr>
              <th className="w-12 px-4 py-3"></th>
              {columns.map((col) => (
                <th
                  key={col.key}
                  className={`${col.width} px-4 py-3 font-semibold cursor-pointer hover:bg-[#153d73] transition-colors`}
                  onClick={() => handleSort(col.key as keyof RecoveryCase)}
                >
                  <div className="flex items-center gap-1">
                    {col.label}
                    {sortConfig?.key === col.key && (
                      <ChevronDown
                        className={`h-3 w-3 transition-transform ${sortConfig.direction === 'asc' ? 'rotate-180' : ''}`}
                      />
                    )}
                  </div>
                </th>
              ))}
            </tr>
          </thead>
          <tbody className="divide-y divide-slate-100">
            {paginatedData.map((row) => {
              const isUrgent = isUrgentRow(row)
              const isExpanded = expandedRowId === row.id
              const taskProgress = getTaskProgress(row.tasks)
              return (
                <Fragment key={row.id}>
                  <tr
                    className={`
                      group transition-all cursor-pointer
                      ${isUrgent ? 'bg-red-50/60 hover:bg-red-50' : 'hover:bg-slate-50'}
                      ${isExpanded ? 'bg-slate-50' : ''}
                    `}
                    onClick={() => toggleRowExpansion(row.id)}
                  >
                    {/* Expand indicator + Urgent flag */}
                    <td className="px-4 py-3">
                      <div className="flex items-center gap-2">
                        {isExpanded ? (
                          <ChevronDown className="h-4 w-4 text-slate-400" />
                        ) : (
                          <ChevronRight className="h-4 w-4 text-slate-400" />
                        )}
                        {isUrgent && (
                          <AlertTriangle className="h-4 w-4 text-red-500" />
                        )}
                      </div>
                    </td>

                    {/* Technician Name + Enterprise ID */}
                    <td className="px-4 py-3">
                      <div className="font-medium text-slate-900">
                        {row.techName}
                      </div>
                      <div className="text-xs text-slate-400 font-mono">
                        {row.enterpriseId}
                      </div>
                    </td>

                    {/* District */}
                    <td className="px-4 py-3 text-slate-600">{row.district}</td>

                    <td className="px-4 py-3 text-slate-600">

                      </Badge>
                    </td>

                    {/* Routing */}
                    <td className="px-4 py-3 text-slate-600">{row.routing}</td>

                    <td className="px-4 py-3">

                        variant={getStatusBadgeVariant(row.status)}

                      >

              disabled={currentPage === 1}

            >
              Next
            </button>
          </div>
        </div>
      )}

      {data.length === 0 && (
        <div className="p-12 text-center text-slate-500">
          No recovery cases found matching your filters.
        </div>
      )}
    </div>
  )
}
```
