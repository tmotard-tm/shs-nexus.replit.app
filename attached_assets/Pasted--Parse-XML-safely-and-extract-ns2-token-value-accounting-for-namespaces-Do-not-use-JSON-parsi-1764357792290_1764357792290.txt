// Parse XML safely and extract ns2:token value, accounting for namespaces
// Do not use JSON parsing per requirement.

pm.test("Parse XML and set collection token", function () {
    // Get raw XML as text
    const xml = pm.response.text();

    // Use DOMParser via Postman sandbox to parse XML
    // Fallback to regex if DOMParser is unavailable (older sandboxes)
    let token = null;

    try {
        // xml2Json is available in Postman tests; it preserves namespaced keys using ':' in property names
        var jsonObject = xml2Json(xml);
        // Find the token regardless of default namespace prefix by checking typical structures
        // The root is likely ns2:TokenResponse with default namespace aliases
        // xml2Json keeps namespace prefixes in the key names. We'll search recursively for a key ending with ':token'.
        function findToken(obj) {
            if (!obj || typeof obj !== 'object') return null;
            // Direct hit
            if (Object.prototype.hasOwnProperty.call(obj, 'ns2:token')) {
                // xml2Json wraps text nodes as string values
                return obj['ns2:token'];
            }
            // Some XMLs may drop prefix during conversion, try plain 'token'
            if (Object.prototype.hasOwnProperty.call(obj, 'token')) {
                return obj['token'];
            }
            // Recurse
            for (const k of Object.keys(obj)) {
                const v = obj[k];
                const found = findToken(v);
                if (found) return found;
            }
            return null;
        }

        token = findToken(jsonObject);
    } catch (e) {
        // As a last resort, use a conservative regex to capture <ns2:token>...</ns2:token>
        const m = xml.match(/<ns2:token>([\s\S]*?)<\/ns2:token>/i);
        if (m) token = m[1].trim();
    }

    pm.expect(token, 'ns2:token should be present in XML').to.be.a('string').and.not.empty;

    // Save to collection variable
    pm.collectionVariables.set('token', token);
});
