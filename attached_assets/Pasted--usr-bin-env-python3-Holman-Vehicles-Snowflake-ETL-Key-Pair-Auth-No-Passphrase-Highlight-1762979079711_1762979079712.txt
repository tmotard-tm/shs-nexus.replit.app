"""
Holman Vehicles → Snowflake ETL (Key-Pair Auth, No Passphrase)

Highlights:
- Uses lesseeCodes + statusCodes (comma list) for the vehicles basic-query.
- Two-pass strategy:
  1) Combined paged run for non-3 statuses (e.g., 0,1,2).
  2) Dedicated paged run for sold (status 3) with soldDateCode=5.
- Deltas via lastChangeRecordID (primary) with lastChangeDate fallback.
- Snowflake MERGE is de-duplicated (QUALIFY ROW_NUMBER) and uses safe casts from VARIANT.
- IMPORTANT: holmanVehicleNumber is passed through EXACTLY AS RETURNED (no padding/stripping).

Required env:
  HOLMAN_CLIENT_ID, HOLMAN_CLIENT_SECRET, HOLMAN_LESSEE_CODES (e.g., "2B56")
  SNOWFLAKE_ACCOUNT, SNOWFLAKE_DATABASE, SNOWFLAKE_SCHEMA,
  SNOWFLAKE_WAREHOUSE, SNOWFLAKE_USER, SF_PRIVATE_KEY_PATH
Optional:
  HOLMAN_ENV=prod|qa (default prod), HOLMAN_STATUS_CODES (default "1,2,3"),
  HOLMAN_SOLD_DATE_CODE (default "5"), SF_ROLE (default "SCIENTIST")

# ---------- Helpers ----------
def load_private_key_der(path: str) -> bytes:
    with open(path,"rb") as f:
        data = f.read()
    try:
        key = serialization.load_pem_private_key(data, password=None, backend=default_backend())
    except ValueError:
        key = serialization.load_der_private_key(data, password=None, backend=default_backend())
    return key.private_bytes(
        encoding=serialization.Encoding.DER,
        format=serialization.PrivateFormat.PKCS8,
        encryption_algorithm=serialization.NoEncryption(),
    )

def connect_snowflake() -> sf.SnowflakeConnection:
    pk_der = load_private_key_der(SF_PRIVATE_KEY_PATH)
    conn = sf.connect(
        account=SF_ACCOUNT,
        user=SF_USER,
        private_key=pk_der,
        role=SF_ROLE,
        warehouse=SF_WAREHOUSE,
        database=SF_DATABASE,
        schema=SF_SCHEMA,
        client_session_keep_alive=True,
    )
    return conn

def get_access_token() -> str:
    data={"grant_type":"client_credentials","client_id":HOLMAN_CLIENT_ID,"client_secret":HOLMAN_CLIENT_SECRET}
    for attempt in range(5):
        try:
            r=SESSION.post(TOKEN_URL,data=data,timeout=30)
            r.raise_for_status()
            return r.json()["access_token"]
        except Exception as e:
            wait = min(60, 2**attempt)
            logger.warning("Auth failed (%s). Retrying in %ss", e, wait)
            time.sleep(wait)
    raise RuntimeError("Failed to obtain access token after retries.")

def iso_to_holman_datetime(iso: str) -> Optional[str]:
    """Convert ISO 8601 to 'MM/DD/YYYY hh:mm:ss AM/PM' in UTC."""
    if not iso:
        return None
    try:
        dt = datetime.fromisoformat(iso.replace("Z","+00:00")).astimezone(timezone.utc)
        return dt.strftime("%m/%d/%Y %I:%M:%S %p")
    except Exception:
        return None

def fetch_page(token: str, lessee: str, page: int, status_codes: List[str], sold_date_code: Optional[str],
               last_change_record_id: Optional[int], last_change_date: Optional[str]) -> dict:
    """
    Vehicles basic-query for one page. Primary attempt uses:
      - lesseeCodes (plural)
      - statusCodes (plural, comma-joined)
    If that fails, try param-name / URL variants:
      - lesseeCode
      - statusCode (comma-joined)
      - base URL with trailing slash
    """
    bases = [
        f"{HOLMAN_API_BASE}/CustomerDataAPI/vehicles/basic-query",
        f"{HOLMAN_API_BASE}/CustomerDataAPI/vehicles/basic-query/",
    ]
    headers = {
        "Authorization": f"Bearer {token}",
        "Accept": "application/json",
        "x-correlation-id": f"veh-{lessee}-{','.join(status_codes)}-{page}-{int(time.time()*1000)}",
    }

    def _delta_params():
        dp = {}
        if last_change_record_id:
            dp["lastChangeRecordID"] = str(last_change_record_id)
        elif last_change_date:
            lcd = iso_to_holman_datetime(last_change_date)
            if lcd:
                dp["lastChangeDate"] = lcd
        return dp

    tried = []
    first_err = None
    joined = ",".join(status_codes)

    attempts = []
    for base in bases:
        attempts.append((base, "lesseeCodes", "statusCodes"))
        attempts.append((base, "lesseeCodes", "statusCode"))
        attempts.append((base, "lesseeCode",  "statusCodes"))
        attempts.append((base, "lesseeCode",  "statusCode"))

    for base, lkey, skey in attempts:
        params = {
            lkey: lessee,
            "pageNumber": page,
            "pageSize": PAGE_SIZE,
            skey: joined,
        }
        if joined == "3" and sold_date_code:
            params["soldDateCode"] = sold_date_code
        params.update(_delta_params())

        try:
            r = SESSION.get(base, headers=headers, params=params, timeout=60)
            if r.status_code >= 400:
                if not first_err:
                    first_err = (r.status_code, r.text, base, params.copy())
                r.raise_for_status()
            return r.json()
        except requests.HTTPError:
            tried.append((base, params))

    if first_err:
        code, body, base, params = first_err
        logger.error("Holman API error %s: %s (first attempt: %s params=%s). Tried %d variants.",
                     code, body, base, params, len(tried))
    else:
        logger.error("Holman API request failed with no response body. Tried %d variants.", len(tried))
    raise requests.HTTPError("Holman basic-query failed for all param variants")

def fetch_all_for_lessee(token: str, lessee: str, last_change_record_id: Optional[int], last_change_date: Optional[str]) -> Iterable[dict]:
    to_pull = [s for s in HOLMAN_STATUS_CODES]
    non3 = [s for s in to_pull if s != "3"]
    has3 = "3" in to_pull

    # First: combined pass for non-3 statuses (e.g., 0,1,2)
    if non3:
        page = 1
        while True:
            payload = fetch_page(token, lessee, page, non3, None, last_change_record_id, last_change_date)
            items = (payload or {}).get("items") or []
            for it in items:
                yield it
            pi = (payload or {}).get("pageInfo") or {}
            total_pages = int(pi.get("totalPages") or 0)
            if page >= max(total_pages, 1) or not items:
                break
            page += 1

    # Second: dedicated pass for status 3 (sold) with soldDateCode
    if has3:
        page = 1
        while True:
            payload = fetch_page(token, lessee, page, ["3"], HOLMAN_SOLD_DATE_CODE, last_change_record_id, last_change_date)
            items = (payload or {}).get("items") or []
            for it in items:
                yield it
            pi = (payload or {}).get("pageInfo") or {}
            total_pages = int(pi.get("totalPages") or 0)
            if page >= max(total_pages, 1) or not items:
                break
            page += 1

UPSERT_CHECKPOINT_SQL = """
merge into {schema}.HOLMAN_VEHICLES_CHECKPOINTS t
using (select %(lessee)s as lessee_code, %(lcd)s as last_change_date, %(lcrid)s as last_change_record_id) s
on t.lessee_code = s.lessee_code
when matched then update set last_change_date = s.last_change_date, last_change_record_id = s.last_change_record_id
when not matched then insert (lessee_code, last_change_date, last_change_record_id) values (s.lessee_code, s.last_change_date, s.last_change_record_id);

def exec_multi(cur: sf.cursor.SnowflakeCursor, sql_text: str):
    for stmt in [s.strip() for s in sql_text.split(";") if s.strip()]:
        logger.info("Executing: %s...", " ".join(stmt.split()[:4]))
        cur.execute(stmt)

def main():
    token = get_access_token()
    logger.info("Auth OK; HOLMAN_ENV=%s", HOLMAN_ENV)

    conn = connect_snowflake()
    cur = conn.cursor()

    # explicit context
    cur.execute(f"USE ROLE {SF_ROLE}")
    cur.execute(f"USE WAREHOUSE {SF_WAREHOUSE}")
    cur.execute(f"USE DATABASE {SF_DATABASE}")
    cur.execute(f"USE SCHEMA {SF_SCHEMA}")

    # ensure schema & tables
    exec_multi(cur, DDL_SQL.format(schema=SF_SCHEMA))

    # migration: add last_change_record_id if missing
    cur.execute(f"alter table if exists {SF_SCHEMA}.HOLMAN_VEHICLES_CHECKPOINTS add column if not exists last_change_record_id number")

    for lessee in HOLMAN_LESSEE_CODES:
        # load checkpoint
        cur.execute(f"select last_change_date, last_change_record_id from {SF_SCHEMA}.HOLMAN_VEHICLES_CHECKPOINTS where lessee_code=%s", (lessee,))
        row = cur.fetchone()
        last_change_date = row[0] if row and row[0] else None
        last_change_record_id = int(row[1]) if row and row[1] is not None else None
        logger.info("Lessee %s: starting (last_change_record_id=%s, last_change_date=%s)", lessee, last_change_record_id, last_change_date or "<none>")

        # pull items
        max_id = last_change_record_id or 0
        max_date = last_change_date

        batch = []
        for it in fetch_all_for_lessee(token, lessee, last_change_record_id, last_change_date):
            # IMPORTANT: DO NOT normalize holmanVehicleNumber here—keep EXACT from API
            batch.append((lessee, json.dumps(it)))
            lcrid = it.get("lastChangeRecordId")
            if isinstance(lcrid, int) and lcrid > max_id:
                max_id = lcrid
            lcd = it.get("lastChangeDate")
            if lcd and (not max_date or lcd > max_date):
                max_date = lcd

            if len(batch) >= 1000:
                cur.executemany(
                    f"insert into {SF_SCHEMA}.HOLMAN_VEHICLES_RAW(lessee_code, payload) "
                    f"select column1, parse_json(column2) from values(%s,%s)",
                    batch
                )
                conn.commit()
                batch = []

        if batch:
            cur.executemany(
                f"insert into {SF_SCHEMA}.HOLMAN_VEHICLES_RAW(lessee_code, payload) "
                f"select column1, parse_json(column2) from values(%s,%s)",
                batch
            )
            conn.commit()

        # merge curated
        cur.execute(MERGE_SQL.format(schema=SF_SCHEMA, lessee=lessee))
        conn.commit()

        # upsert checkpoint
        cur.execute(
            UPSERT_CHECKPOINT_SQL.format(schema=SF_SCHEMA),
            {"lessee": lessee, "lcd": max_date, "lcrid": max_id if max_id else None}
        )
        conn.commit()

    cur.close(); conn.close()
    logger.info("ETL complete.")

if __name__ == "__main__":
    main()
