Blocking Logic
function getToolsTaskStatus(offboardingId, truckNumber) {
    const isByov = truckNumber.startsWith('88');
    
    if (isByov) {
        return {
            status: 'ROUTING_RECEIVED',
            routingPath: 'BYOV',
            blockedActions: []
        };
    }
    
    const fleetTask = getFleetRoutingTask(offboardingId);
    
    if (!fleetTask || fleetTask.status !== 'complete') {
        return {
            status: 'AWAITING_ROUTING',
            routingPath: null,
            blockedActions: ['issue_qr_codes', 'coordinate_audit']
        };
    }
    
    return {
        status: 'ROUTING_RECEIVED',
        routingPath: fleetTask.routingDecision,
        blockedActions: []
    };
}
