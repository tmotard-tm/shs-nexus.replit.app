def _normalize_sf_account(val: str) -> str:
    """
    Accepts either an account locator (xy12345 or xy12345.us-east-1) or a full hostname
    (xy12345.us-east-1.snowflakecomputing.com) and returns a connector-friendly value.
    """
    if not val:
        return val
    v = val.strip().lower()
    if v.startswith("https://"):
        v = v[8:]
    if v.startswith("http://"):
        v = v[7:]
    suffixes = (
        ".snowflakecomputing.com",
        ".privatelink.snowflakecomputing.com",
    )
    for s in suffixes:
        if v.endswith(s):
            v = v[: -len(s)]
            break
    return v.strip().strip("/")

# ---------------- Helpers ----------------

def load_private_key_der_no_passphrase(pem_path: str) -> bytes:
    with open(pem_path, "rb") as f:
        pem = f.read()
    pkey = serialization.load_pem_private_key(pem, password=None, backend=default_backend())
    # Snowflake connector expects DER-encoded PKCS8
    return pkey.private_bytes(
        encoding=serialization.Encoding.DER,
        format=serialization.PrivateFormat.PKCS8,
        encryption_algorithm=serialization.NoEncryption(),
    )

def sf_connect():
    pkb = load_private_key_der_no_passphrase(SF_KEY_PATH)
    conn = snowflake.connector.connect(
        account=SF_ACCOUNT,      # e.g., xy12345 or xy12345.us-east-1
        user=SF_USER,
        role=SF_ROLE,
        warehouse=SF_WAREHOUSE,
        database=SF_DATABASE,
        schema=SF_SCHEMA,
        private_key=pkb,
    )
    return conn

def get_token() -> str:
    data = {
        "grant_type": "client_credentials",
        "client_id": CLIENT_ID,
        "client_secret": CLIENT_SECRET,
    }
    headers = {"Content-Type": "application/x-www-form-urlencoded"}
    r = requests.post(TOKEN_URL, data=data, headers=headers, timeout=60)
    r.raise_for_status()
    j = r.json()
    tok = j.get("access_token")
    if not tok:
        raise RuntimeError(f"No access_token in token response: {j}")
    return tok

def norm_cell(v: Optional[Any]):
    """Normalize general stringish cells to None when blank/NULL text."""
    if v is None:
        return None
    s = str(v).strip()
    if s == "" or s.lower() == "null":
        return None
    return s

def hvn_str(val) -> Optional[str]:
    """Return HVN exactly as provided; no padding, no stripping."""
    if val is None:
        return None
    s = str(val).strip()
    if s == "" or s.lower() == "null":
        return None
    return s  # <-- do not modify

def row_to_payload(row: Dict[str, Any]) -> Dict[str, Any]:
    # Inject static lesseeCode and include SQL-provided fields
    payload = {"lesseeCode": LESSEE_CODE}
    payload["holmanVehicleNumber"] = hvn_str(row.get("holmanVehicleNumber"))
    for k in FIELDS_FROM_SQL:
        if k == "holmanVehicleNumber":
            continue
        payload[k] = norm_cell(row.get(k))
    return payload

def post_submit(token: str, item: Dict[str, Any]) -> requests.Response:
    headers = {
        "Authorization": f"Bearer {token}",
        "Content-Type": "application/json",
        "Accept": "application/json",
    }
    # Holman submit expects an array, even for single record
    return requests.post(SUBMIT_URL, headers=headers, json=[item], timeout=120)

def poll_status(token: str, submission_id: Optional[str], timeout_sec: int) -> Dict[str, Any]:
    headers = {"Authorization": f"Bearer {token}"}
    params = {"submissionId": submission_id} if submission_id else None
    start = time.time()
    while True:
        r = requests.get(STATUS_URL, headers=headers, params=params, timeout=60)
        r.raise_for_status()
        try:
            j = r.json()
        except Exception:
            j = {"raw_text": r.text}
        result_text = (j.get("result_text") or j.get("resultText") or "").strip().lower()
        if result_text == "complete":
            return j
        if time.time() - start > timeout_sec:
            return {"timeout": True, "last_status": j}
        time.sleep(5)

def sf_log(
    cur,
    endpoint: str,
    http_status: Optional[int],
    submission_id: Optional[str],
    lessee_code: Optional[str],
    holman_vehicle_number: Optional[str],
    duration_ms: Optional[int],
    request_obj: Dict[str, Any],
    response_obj: Dict[str, Any],
    error_text: Optional[str],
):
    sql = """
        INSERT INTO PARTS_SUPPLYCHAIN.FLEET.HOLMAN_TPMS_UPDATES_API_LOG
        (ENDPOINT, HTTP_STATUS, SUBMISSION_ID, LESSEE_CODE, HOLMAN_VEHICLE_NUMBER,
         DURATION_MS, REQUEST_JSON, RESPONSE_JSON, ERROR_TEXT)
        SELECT %s, %s, %s, %s, %s, %s, PARSE_JSON(%s), PARSE_JSON(%s), %s
    """
    cur.execute(
        sql,
        (
            endpoint,
            http_status,
            submission_id,
            lessee_code,
            holman_vehicle_number,
            duration_ms,
            json.dumps(request_obj),
            json.dumps(response_obj),
            error_text,
        ),
    )

# ---------------- Main ----------------

def main():
    _debug_env()

    # Connect to Snowflake
    conn = sf_connect()
    cur = conn.cursor()
    try:
        # Create local log table (optional safety)
        cur.execute(CREATE_LOG_TABLE_SQL)

        # Pull rows to update
        print("Running source query...")
        cur.execute(SQL_SOURCE)
        cols = [c[0] for c in cur.description]
        rows = [dict(zip(cols, r)) for r in cur.fetchall()]

        if not rows:
            print("No rows returned by the SQL.")
            return

        # Build payloads (always include listed fields; blanks => null)
        records_raw = [row_to_payload(r) for r in rows]

        # Filter invalid identifiers (and show count)
        records: List[Dict[str, Any]] = []
        skipped = 0
        for rec in records_raw:
            hvn = rec.get("holmanVehicleNumber")
            if hvn is None or str(hvn).strip() == "":
                skipped += 1
                print("SKIP: missing holmanVehicleNumber for a row; not sending this record.")
                continue
            records.append(rec)

        print(f"Prepared {len(records)} records. Skipped {skipped} due to missing identifier.")
        if not records:
            print("No valid records to submit.")
            return

        # Preview first payload
        print("Sample payload (first record):")
        try:
            print(json.dumps(records[0], indent=2))
        except Exception:
            print(str(records[0])[:500])

        # Prepare Holman token
        token = get_token()
        print(f"Auth OK to Holman ({ENV}).")

        # Submit one-by-one (safer; clearer errors)
        for idx, rec in enumerate(records, start=1):
            lessee = LESSEE_CODE
            hvn    = rec.get("holmanVehicleNumber")

            t0 = time.time()
            http_status = None
            submission_id = None
            req_obj = rec
            resp_obj: Dict[str, Any] = {}
            error_text = None

            try:
                resp = post_submit(token, rec)
                http_status = resp.status_code
                try:
                    resp_obj = resp.json()
                except Exception:
                    resp_obj = {"raw_text": resp.text}

                if isinstance(resp_obj, dict):
                    for k in ("submissionId", "submission_id", "id"):
                        if k in resp_obj:
                            submission_id = str(resp_obj[k])
                            break

                if POLL_STATUS:
                    status = poll_status(token, submission_id, timeout_sec=STATUS_TIMEOUT_SEC)
                    resp_obj["status_poll"] = status

                resp.raise_for_status()

            except Exception as e:
                error_text = f"{type(e).__name__}: {e}"

            duration_ms = int((time.time() - t0) * 1000)

            # Log to Snowflake (your target table)
            sf_log(
                cur=cur,
                endpoint=SUBMIT_URL,
                http_status=http_status,
                submission_id=submission_id,
                lessee_code=lessee,
                holman_vehicle_number=hvn,
                duration_ms=duration_ms,
                request_obj=req_obj,
                response_obj=resp_obj,
                error_text=error_text
            )
            conn.commit()

            print(f"Sent {idx}/{len(records)} hvn={hvn}: status={http_status}, submission_id={submission_id}, ms={duration_ms}")

    finally:
        try:
            cur.close()
        except Exception:
            pass
        try:
            conn.close()
        except Exception:
            pass

if __name__ == "__main__":
    try:
        main()
        print("\nDone.")
    except SystemExit:
        raise
    except Exception:
        print("\nERROR:\n")
        traceback.print_exc()
    finally:
        # Modified: no final input/pause; exit immediately
        pass
